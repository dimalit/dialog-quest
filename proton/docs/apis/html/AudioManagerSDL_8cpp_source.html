00064 bool AudioManagerSDL::Init()
00065 {
00066         // we'll use SDL_Mixer to do all our sound playback. 
00067         // It's a simple system that's easy to use. The most 
00068         // complicated part about using SLD_Mixer is initting it. So 
00069         // let's talk about that. 
00070         // this is the function definition:
00071         // Mix_OpenAudio(int frequency, Uint16 format, int channels, int chunksize);
00072         // here's what it wants:
00073         // 
00074         // frequency: this is the sample rate you want the audio to
00075         // play at. On the Palm, things are optimized for 44100 samples per
00076         // second (CD quality). Though you can send in whatever sample rate you like.
00077         // 
00078         // format: For the Palm, you should use AUDIO_S16
00079         // 
00080         // channels: 1 for mono. 2 for stereo
00081         // 
00082         // chunksize: How big one audio buffer is, in bytes.
00083         // 
00084         // this example has the recommended settings for initting the mixer:
00085 
00086         int rate = 44100;
00087         Uint16 format = AUDIO_S16;
00088         int channels = 2;
00089         int bufferSize = 1024;
00090         
00091         if ( Mix_OpenAudio(rate, format, channels, bufferSize) )
00092         {
00093                 // we had an error opening the audio
00094                 LogMsg("unable to open Audio! Reason: %s\n", Mix_GetError());
00095                 return false;
00096 
00097         }
00098 
00099         // SDL mixer requires that you specify the maximum number of simultaneous
00100         // sounds you will have. This is done through the function Mix_AllocateChannels. 
00101         // We'll need an answer for that. A channel has very little overhead,
00102         // so we'll arbitrarily allocate 16 channels (even though we're only using
00103         // 1 actual sound channel). Among other things, we need a channel per unique
00104         // simultaneous sound playback. So if we play the same sound 4 times, and have
00105         // 4 copies of the sound playing from different start times, we need four channels.
00106         // Note - streamed music does not consume a channel. Only sound playback does.
00107         // So why allocate more than we need? Because it's such a small overhead, and
00108         // later we don't need to come and revisit this every time we add a new sound. 
00109         Mix_AllocateChannels(NUM_CHANNELS);
00110 
00111         Mix_HookMusicFinished(musicFinishedCallback);
00112 
00113         return true;
00114 }
